## POJ3279

枚举状态很容易想到，反转条件是上方格子为黑，因为一个格子下方的格子是最后一个可能更新它的格子，最后检查末行就行，另外注意这题要维护最小答案

```c++
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<stack>
#include<queue>
#include<map>
using namespace std;
#define rep(i,n) for(int (i)=0;(i)<(int)(n);++(i))
#define rel(i,n) for(int (i)=1;(i)<=(int)(n);++(i))
typedef long long LL;
const int INF=0x3f3f3f3f;
int pic[20][20];
int fli[20][20];
int use[20][20];
int pri[20][20];
int m,n,ans,cnt;
void flip(int i,int j)
{
    //cout<<i<<' '<<j<<endl;
    cnt++;
    use[i][j]=1;
    fli[i][j]=!fli[i][j];
    fli[i-1][j]=!fli[i-1][j];
    fli[i+1][j]=!fli[i+1][j];
    fli[i][j-1]=!fli[i][j-1];
    fli[i][j+1]=!fli[i][j+1];
}
void solve(int u)
{
    for(int i=1;i<=m;i++)
    {
        for(int j=1;j<=n;j++)
        fli[i][j]=pic[i][j];
    }
    memset(use,0,sizeof(use));
    cnt=0;
    for(int j=0;j<n;j++)
        if((u>>j)&1) flip(1,j+1);
    for(int i=2;i<=m;i++)
        for(int j=1;j<=n;j++)
        if(fli[i-1][j]) flip(i,j);
    for(int j=1;j<=n;j++) if(fli[m][j]) return;
    if(ans>cnt)
    {
        ans=cnt;
        //cout<<cnt<<endl;
        for(int i=1;i<=m;i++)
        {
            for(int j=1;j<=n;j++)
            pri[i][j]=use[i][j];
        }
    }
}
int main()
{
    //freopen("test.txt","r",stdin);
    while(scanf("%d%d",&m,&n)==2)
    {
        for(int i=1;i<=m;i++)
            for(int j=1;j<=n;j++)
            scanf("%d",&pic[i][j]);
        ans=INF;
        int up=1<<n;
        for(int u=0;u<up;u++) solve(u);
        if(ans==INF) printf("IMPOSSIBLE\n");
        else
        for(int i=1;i<=m;i++)
        {
            for(int j=1;j<=n;j++)
            {
                printf("%d",pri[i][j]);
                if(j!=n) printf(" ");
            }
            printf("\n");
        }

    }
    return 0;
}
```

## HDU5723

### 题意

有n（n<=100000）个点，现在需要建点之间的m（m<=1000000）条边，每条边长wi（wi<=1000000），没有重复的wi，每条路的费用和边长相等。希望花费最小，且建完后图是连通图。现在随机选两个点作为起点和终点，求最小花费和两点间距离的最小期望值（保留两位小数）。

### 题解

边权不重复，因此最小生成树是唯一的，求最小花费，就是求最小生成树。
求任意两点间距离的期望值，就是求最小生成树上任两点间距离的平均值。
边在所有路径上的总使用次数\*边权是含有这条边的总距离，对所有边求和再除以总路径数即可，方法是求最小生成树时给结点建树，把去结点压进来树，来结点压进去树，最后dfs统计每个结点儿子的总距离。

### 代码

```c++
#include<iostream>
#include<string.h>
#include<vector>
#include<queue>
#include<algorithm>
#include<stdio.h>
using namespace std;
const int N=1000010;
typedef long long LL;
int f[N];
struct Node
{
    int v,d;
};
vector<Node>tu[N];
double dp[N];
int sum[N];
void init(int n)
{
    for(int i=0;i<=n;i++)
        f[i]=i,tu[i].clear();
}
int fa(int x)
{
    if(x!=f[x])
        return f[x]=fa(f[x]);
    return f[x];
}
struct hh
{
    int u,v,d;
}xx[N];
bool cmp(hh a,hh b)
{
    return a.d<b.d;
}
void dfs(int cur,int ft,int n)
{
    sum[cur]=1;
    for(int i=0,sz=tu[cur].size();i<sz;i++)
    {
        int sn=tu[cur][i].v;
        int l=tu[cur][i].d;
        if(sn==ft) continue;
        dfs(sn,cur,n);
        sum[cur]+=sum[sn];
        double tmp=((double)sum[sn]*(double)(n-sum[sn]))*(double)l;
        dp[cur]+=dp[sn]+tmp;
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,m;
        scanf("%d%d",&n,&m);
        init(n);
        for(int i=0;i<m;i++)
            scanf("%d%d%d",&xx[i].u,&xx[i].v,&xx[i].d);
        sort(xx,xx+m,cmp);
        LL ans=0;
        int k=0;
        for(int i=0;i<m;i++)
        {
            int f1=fa(xx[i].u),f2=fa(xx[i].v);
            if(f1!=f2)
            {
                Node x1,x2;
                x1.v=xx[i].v,x1.d=xx[i].d;
                x2.v=xx[i].u,x2.d=xx[i].d;
                tu[xx[i].u].push_back(x1);
                tu[xx[i].v].push_back(x2);
                f[f1]=f[f2];
                ans+=(LL)xx[i].d;
                k++;
            }
            if(k==n-1)
                break;
        }
        memset(sum,0,sizeof(sum));
        memset(dp,0,sizeof(dp));
        dfs(1,0,n);
        double tt=((double)n*(double)(n-1))/2.0;
        printf("%I64d %.2f\n",ans,dp[1]/tt);
    }
    return 0;
}
```
